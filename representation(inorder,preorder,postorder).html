                #include
                struct node{
                    int data;
                    struct node* left;
                    struct node* right;
                };
                void preorder(struct node* name)
                {
                    if(name!=NULL)
                    {
                        printf("%d\n",name->data);
                        preorder(name->left);
                        preorder(name->right);
                    }
                }
                struct node* function(int data)
                {
                        struct node* n;
                        n=(struct node*)malloc(sizeof(struct node));
                        n->data=data;
                        n->left=NULL;
                        n->right=NULL;
                    }
                main(){
                    struct node* p=function(123);

                    struct node* p2=function(45);
                    struct node*p3=function(12);

                    p->left=p1;
                    p->right=p2;
                    p1->left=p3;
                    p1->right=p4;
                    preorder(p);
                }

2.traversing in binary tree(post order)

            #include
                #include
                struct node{
                    int data;
                    struct node* left;
                    struct node* right;
                };
                struct node* creatbinary(int data){
                    struct node* n=(struct node*)malloc(sizeof(struct node));
                    n->data=data;
                    n->left=NULL;
                    n->right=NULL;
                    return n;
                }
                void postorder(struct node* root){
                    if(root!=NULL)
                    {
                        postorder(root->left);
                        postorder(root->right);
                        printf("%d\n",root->data);
                    }
                }
                main()
                {
                    struct node* p=creatbinary(12);
                    struct node* p1=creatbinary(56);
                    struct node* p2=creatbinary(45);
                    struct node* p3=creatbinary(34);
                    struct node* p4=creatbinary(23);
                    p->left=p1;
                    p->right=p2;
                    p2->left=p3;
                    p2->right=p4;
                    postorder(p);
                }
