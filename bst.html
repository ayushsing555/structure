                  void insertion(struct node* root,int key)
                  {
                      struct node* prev=NULL;
                      while(root!=NULL)
                      {
                          prev=root;
                          if(key==root->data){		  
                            printf("not possible insertion bcz it will be duplicated");
                            return; 
                      }
                           if(key>root->data)
                             root=root->right;
                            else
                              root=root->left;
                      }
                    struct node* newtree=function(key);
                       if(key>prev->data)
                          prev->right=newtree;
                       else
                          prev->left=newtree;
                  }
